const filterTop = ref(false)
const sortByTime = ref(false)

const displayList = computed(() => {
  let list = [...competitions]
  if (filterTop.value) {
    list = list.filter(c => c.stars === 5)
  }
  if (sortByTime.value) {
    list.sort((a, b) => a.startMonth - b.startMonth)
  }
  return list
})

const activeCount = computed(() => {
  return (filterTop.value ? 1 : 0) + (sortByTime.value ? 1 : 0)
})
